\begin{align*} \int_{0}^{\frac{\pi}{2}}\cos^{2}\theta\,d\theta & =\int_{0}^{1}\frac{1 + \cos2\theta}{2}\,d\theta \\ & = \frac{1}{2}\int_{0}^{\frac{\pi}{2}}(1 + \cos2\theta)\,d\theta \\ & = \frac{1}{2}\left[\theta + \frac{1}{2}\sin2\theta\right]_{0}^{\frac{\pi}{2}} \\ & = \frac{\pi}{4}. \end{align*}

\begin{align*} \iint_{T}(x^{2} + y^{2})\,dA & = \int_{0}^{1}\int_{x}^{2-x}(x^{2} + y^{2})\,dy\,dx \\ & = \int_{0}^{1}\left[x^{2}y + \frac{y^{3}}{3}\right]_{y=x}^{2-x}\,dx \\ & = \int_{0}^{1} \left[x^{2}(2-2x) + \frac{1}{3}[(2-x)^{3} - x^{3}]\right]\,dx \\ & = \left[\frac{2x^{3}}{3} - \frac{1}{2}x^{4} - \frac{(2-x)^{4}}{12} - \frac{x^{4}}{4}\right]_{0}^{1} \\ & = \frac{2}{3} - \frac{1}{2} - \frac{1}{12} - \frac{1}{4} + \frac{16}{12} \\ & = \frac{7}{6} \end{align*}

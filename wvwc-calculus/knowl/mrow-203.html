\begin{align*} \int\frac{1}{2+\cos x}\,dx \amp = \int\frac{2}{2(1+t^{2}) + (1-t^{2})}\,dt\\ \amp = \int\frac{2}{3 + t^{2}}\,dt\\ \amp = \frac{2}{3}\int\frac{1}{1 + (t/\sqrt{3})^{2}}\,dt\\ \amp = \frac{2}{3}\sqrt{3}\arctan(\frac{t}{\sqrt{3}}) + C\\ \amp = \frac{2}{\sqrt{3}}\arctan(\frac{\tan(x/2)}{\sqrt{3}}) + C \end{align*}

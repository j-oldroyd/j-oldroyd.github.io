\begin{align*} \int_{0}^{3}\sqrt{(e^{t}-e^{-t})^{2} + 4}\,dt \amp = \int_{0}^{3}\sqrt{e^{2t}-2 + e^{-2t} + 4}\,dt \\ \amp = \int_{0}^{3}\sqrt{e^{2t} + 2 + e^{-2t}}\,dt \\ \amp = \int_{0}^{3}\sqrt{(e^{t}+e^{-t})^{2}}\,dt \\ \amp = \int_{0}^{3}(e^{t}+e^{-t})\,dt\\ \amp = e^{3}-e^{-3}. \end{align*}

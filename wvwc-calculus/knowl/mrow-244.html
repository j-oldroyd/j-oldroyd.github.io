\begin{align*} \int\frac{du}{(4 + u^{2})^{3/2}} & = \int\frac{2\sec^{2}\theta}{4^{3/2}(1 + \tan^{2}\theta)^{3/2}}\,d\theta \\ & = \frac{1}{4}\int\cos\theta\,d\theta \\ & = \frac{1}{4}\sin\theta + C \\ & = \frac{1}{4}\sin\arctan\frac{u}{2} + C \\ & = \frac{1}{4}\sin\arctan\frac{e^{t}}{2} + C \end{align*}

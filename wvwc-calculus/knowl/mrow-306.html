\begin{align*} 2\int_{-r}^{r}\sqrt{1 + [y']^{2}}\,dx \amp = 2\int_{-r}^{r}\sqrt{1 + \frac{x^{2}}{r^{2} - x^{2}}}\,dx \\ \amp = 2r\int_{-r}^{r}\frac{1}{\sqrt{r^{2} - x^{2}}}\,dx \\ \amp = 2r\int_{-\frac{\pi}{2}}^{\frac{\pi}{2}}\frac{r\cos\theta}{r\cos\theta}\,d\theta \\ \amp = 2\pi r. \end{align*}

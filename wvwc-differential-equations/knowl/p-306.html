\begin{align*} \dv{x_{1}}{t} \amp= 5\cdot0 - 5\frac{x_{1}}{10}\\ \dv{x_{2}}{t} \amp= 5\frac{x_{1}}{10}-5\frac{x_{2}}{7} \end{align*}

or just

\begin{align*} x'_{1} \amp= -\frac{1}{2}x_{1}\\ x'_{2} \amp= \frac{1}{2}x_{1}-\frac{5}{7}x_{2}, \end{align*}

with initial conditions \(x_{1}(0) = 5\) and \(x_{2}(0) = 2\text{.}\)
